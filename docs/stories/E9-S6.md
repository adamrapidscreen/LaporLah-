# E9-S6: Build Flag Button Component and Flag Server Actions

## Description
Create a flag button Client Component that opens a dialog modal for users to flag reports or comments with a reason. Implement the corresponding server actions for inserting flags into the database.

## Dependencies
- Stories: E4-S1 (report detail page), E5-S1 (comments system)
- Files: `src/app/report/[id]/page.tsx`, `src/components/ui/button.tsx`, `src/components/ui/dialog.tsx`, `src/components/ui/input.tsx`

## Tasks
1. `src/components/shared/flag-button.tsx`: Create Client Component that:
   - Renders a flag icon button (lucide `Flag` icon)
   - On click, opens a shadcn Dialog modal
   - Modal contains: title ("Report this [report/comment]"), reason textarea (required, max 500 chars), Cancel and Submit buttons
   - Validates reason is non-empty before submitting
   - Calls `flagReport` or `flagComment` server action based on `type` prop
   - Shows loading state on submit, closes dialog on success
   - Displays error message if action fails
2. `src/lib/actions/flags.ts`: Create server actions file with 'use server' directive containing:
   - `flagReport(reportId: string, reason: string)`: Validate with Zod, check auth, INSERT into flags table with report_id
   - `flagComment(commentId: string, reason: string)`: Validate with Zod, check auth, INSERT into flags table with comment_id
3. `src/lib/validations/flags.ts`: Create Zod schema for flag input (reason: string, min 1, max 500)

## Implementation Notes

### Flag Button Component
```tsx
// src/components/shared/flag-button.tsx
'use client';

import { useState, useTransition } from 'react';
import { Button } from '@/components/ui/button';
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
  DialogTrigger,
} from '@/components/ui/dialog';
import { Textarea } from '@/components/ui/textarea';
import { Flag } from 'lucide-react';
import { flagReport, flagComment } from '@/lib/actions/flags';

interface FlagButtonProps {
  type: 'report' | 'comment';
  targetId: string;
}

export function FlagButton({ type, targetId }: FlagButtonProps) {
  const [open, setOpen] = useState(false);
  const [reason, setReason] = useState('');
  const [error, setError] = useState<string | null>(null);
  const [isPending, startTransition] = useTransition();

  const handleSubmit = () => {
    if (!reason.trim()) {
      setError('Please provide a reason');
      return;
    }
    if (reason.length > 500) {
      setError('Reason must be 500 characters or less');
      return;
    }

    setError(null);
    startTransition(async () => {
      const action = type === 'report' ? flagReport : flagComment;
      const result = await action(targetId, reason.trim());
      if (result?.error) {
        setError(result.error);
      } else {
        setOpen(false);
        setReason('');
      }
    });
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="ghost" size="icon" className="text-muted-foreground hover:text-destructive">
          <Flag className="h-4 w-4" />
        </Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Report this {type}</DialogTitle>
        </DialogHeader>
        <div className="space-y-3">
          <Textarea
            placeholder="Why are you flagging this? (required)"
            value={reason}
            onChange={(e) => setReason(e.target.value)}
            maxLength={500}
            rows={3}
          />
          <p className="text-xs text-muted-foreground text-right">{reason.length}/500</p>
          {error && <p className="text-sm text-destructive">{error}</p>}
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={() => setOpen(false)} disabled={isPending}>
            Cancel
          </Button>
          <Button variant="destructive" onClick={handleSubmit} disabled={isPending}>
            {isPending ? 'Submitting...' : 'Submit Flag'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

### Flag Server Actions
```typescript
// src/lib/actions/flags.ts
'use server';

import { createServerClient } from '@/lib/supabase/server';
import { z } from 'zod';
import { revalidatePath } from 'next/cache';

const flagSchema = z.object({
  reason: z.string().min(1, 'Reason is required').max(500, 'Reason must be 500 characters or less'),
});

export async function flagReport(reportId: string, reason: string) {
  const parsed = flagSchema.safeParse({ reason });
  if (!parsed.success) return { error: parsed.error.errors[0].message };

  const supabase = await createServerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: 'Not authenticated' };

  const { error } = await supabase
    .from('flags')
    .insert({ report_id: reportId, user_id: user.id, reason: parsed.data.reason });

  if (error) return { error: 'Failed to submit flag' };

  revalidatePath('/admin/flagged');
  return { error: null };
}

export async function flagComment(commentId: string, reason: string) {
  const parsed = flagSchema.safeParse({ reason });
  if (!parsed.success) return { error: parsed.error.errors[0].message };

  const supabase = await createServerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: 'Not authenticated' };

  const { error } = await supabase
    .from('flags')
    .insert({ comment_id: commentId, user_id: user.id, reason: parsed.data.reason });

  if (error) return { error: 'Failed to submit flag' };

  revalidatePath('/admin/flagged');
  return { error: null };
}
```
