# E5-S1: Build Comment List and Comment Bubble Components

## Description
Create the comment list Server Component that fetches and displays all comments for a report, and the comment bubble component that renders individual comments with avatar, name, timestamp, and content.

## Dependencies
- Stories: E4-S1 (report detail page), E1-S2 (comments table in database)
- Files: `src/app/report/[id]/page.tsx`, database `comments` table with `users` foreign key

## Tasks
1. `src/components/comments/comment-list.tsx`: Create Server Component that fetches all comments for a given report ID from Supabase, ordered by `created_at ASC` (oldest first), joined with user data. Render each comment using `CommentBubble`. Show "No comments yet" empty state.
2. `src/components/comments/comment-bubble.tsx`: Create presentational component that displays a single comment with commenter avatar (28px), name, relative time ago, and comment body. Style with `bg-secondary`, `rounded-lg`, `px-3 py-2.5`.
3. `src/app/report/[id]/page.tsx`: Integrate `CommentList` into the report detail page below the description section.

## Implementation Notes

### Comment List Component
```tsx
// src/components/comments/comment-list.tsx
import { createServerClient } from '@/lib/supabase/server';
import { CommentBubble } from '@/components/comments/comment-bubble';

interface CommentListProps {
  reportId: string;
}

export async function CommentList({ reportId }: CommentListProps) {
  const supabase = await createServerClient();
  const { data: comments } = await supabase
    .from('comments')
    .select('*, user:users(id, name, avatar_url)')
    .eq('report_id', reportId)
    .order('created_at', { ascending: true });

  if (!comments || comments.length === 0) {
    return (
      <p className="py-8 text-center text-sm text-muted-foreground">
        Tiada komen lagi / No comments yet
      </p>
    );
  }

  return (
    <div className="space-y-3">
      {comments.map((comment) => (
        <CommentBubble key={comment.id} comment={comment} />
      ))}
    </div>
  );
}
```

### Comment Bubble Component
```tsx
// src/components/comments/comment-bubble.tsx
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar';
import { formatTimeAgo } from '@/lib/utils';

interface Comment {
  id: string;
  content: string;
  created_at: string;
  user: {
    id: string;
    name: string;
    avatar_url: string | null;
  };
}

interface CommentBubbleProps {
  comment: Comment;
}

export function CommentBubble({ comment }: CommentBubbleProps) {
  return (
    <div className="flex gap-2.5">
      <Avatar className="h-7 w-7 shrink-0">
        <AvatarImage src={comment.user.avatar_url ?? undefined} alt={comment.user.name} />
        <AvatarFallback className="text-xs">
          {comment.user.name.charAt(0).toUpperCase()}
        </AvatarFallback>
      </Avatar>
      <div className="rounded-lg bg-secondary px-3 py-2.5">
        <div className="flex items-baseline gap-2">
          <span className="text-sm font-medium">{comment.user.name}</span>
          <span className="text-xs text-muted-foreground">
            {formatTimeAgo(comment.created_at)}
          </span>
        </div>
        <p className="mt-1 text-sm text-foreground">{comment.content}</p>
      </div>
    </div>
  );
}
```

### Time Ago Utility
```tsx
// Addition to src/lib/utils.ts (or new utility)
export function formatTimeAgo(dateString: string): string {
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now.getTime() - date.getTime();
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);

  if (diffMins < 1) return 'baru sahaja'; // just now
  if (diffMins < 60) return `${diffMins}m lalu`;
  if (diffHours < 24) return `${diffHours}j lalu`;
  if (diffDays < 7) return `${diffDays}h lalu`;
  return date.toLocaleDateString('ms-MY');
}
```
